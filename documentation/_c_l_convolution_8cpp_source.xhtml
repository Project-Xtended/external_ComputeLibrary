55 template <unsigned int matrix_size>
56 void CLConvolutionSquare<matrix_size>::configure(ICLTensor *input, ICLTensor *output, const int16_t *conv, uint32_t scale, BorderMode border_mode,
57  uint8_t constant_border_value)
58 {
59  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::U8);
60  ARM_COMPUTE_ERROR_ON(conv == nullptr);
61  std::array<int16_t, matrix_size> conv_col{ 0 };
62  std::array<int16_t, matrix_size> conv_row{ 0 };
63  _is_separable = separate_matrix(conv, conv_col.data(), conv_row.data(), matrix_size);
64 
65  if(_is_separable)
66  {
67  std::pair<DataType, DataType> type_pair = data_type_for_convolution(conv_col.data(), conv_row.data(), matrix_size);
68  _tmp.allocator()->init(TensorInfo(input->info()->tensor_shape(), 1, type_pair.first));
69 
70  // Manage intermediate buffers
71  _memory_group.manage(&_tmp);
72 
73  if(scale == 0)
74  {
75  scale = calculate_matrix_scale(conv, matrix_size);
76  }
77 
78  _kernel_hor.configure(input, &_tmp, conv_row.data(), border_mode == BorderMode::UNDEFINED);
79  _kernel_vert.configure(&_tmp, output, conv_col.data(), scale, border_mode == BorderMode::UNDEFINED, type_pair.second);
80  _border_handler.configure(input, _kernel_hor.border_size(), border_mode, PixelValue(constant_border_value));
81 
82  // Allocate intermediate buffer
83  _tmp.allocator()->allocate();
84  }
85  else
86  {
87  _kernel.configure(input, output, conv, scale, border_mode == BorderMode::UNDEFINED);
88  _border_handler.configure(input, _kernel.border_size(), border_mode, PixelValue(constant_border_value));
89  }
90 }
