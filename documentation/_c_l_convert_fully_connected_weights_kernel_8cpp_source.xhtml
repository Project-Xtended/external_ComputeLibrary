40 void CLConvertFullyConnectedWeightsKernel::configure(const ICLTensor *input, ICLTensor *output, const TensorShape &original_input_shape,
41  DataLayout data_layout)
42 {
43  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
44 
45  // Output tensor auto initialisation if not yet initialized
46  auto_init_if_empty(*output->info(), *input->info()->clone());
47 
48  ARM_COMPUTE_ERROR_THROW_ON(CLConvertFullyConnectedWeightsKernel::validate(input->info(), output->info(), original_input_shape, data_layout));
49 
50  _input = input;
51  _output = output;
52 
53  const DataLayout input_data_layout = (data_layout == DataLayout::NCHW) ? DataLayout::NHWC : DataLayout::NCHW;
54 
55  const int width_idx = get_data_layout_dimension_index(input_data_layout, DataLayoutDimension::WIDTH);
56  const int height_idx = get_data_layout_dimension_index(input_data_layout, DataLayoutDimension::HEIGHT);
57  const int channel_idx = get_data_layout_dimension_index(input_data_layout, DataLayoutDimension::CHANNEL);
58 
59  const unsigned int num_elems_per_input_plane = original_input_shape[width_idx] * original_input_shape[height_idx];
60  const unsigned int num_channels = original_input_shape[channel_idx];
61 
62  const unsigned int factor_1 = (data_layout == DataLayout::NCHW) ? num_elems_per_input_plane : num_channels;
63  const unsigned int factor_2 = (data_layout == DataLayout::NCHW) ? num_channels : num_elems_per_input_plane;
64 
65  // Set build options
66  CLBuildOptions build_opts;
67  build_opts.add_option("-DDATA_TYPE=" + get_cl_unsigned_type_from_element_size(input->info()->element_size()));
68  build_opts.add_option("-DFACTOR_1=" + support::cpp11::to_string(factor_1));
69  build_opts.add_option("-DFACTOR_2=" + support::cpp11::to_string(factor_2));
70 
71  // Create kernel
72  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("convert_fc_weights", build_opts.options()));
73 
74  // Configure kernel window
75  Window win = calculate_max_window(*input->info(), Steps());
76  ICLKernel::configure_internal(win);
77 }

79 Status CLConvertFullyConnectedWeightsKernel::validate(const ITensorInfo *input, const ITensorInfo *output, const TensorShape &original_input_shape,
80  DataLayout data_layout)
81 {
82  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
83  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
84  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
85  ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() != 2);
86  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(1) != original_input_shape.total_size_lower(3));
87  ARM_COMPUTE_RETURN_ERROR_ON(data_layout == DataLayout::UNKNOWN);
88 
89  // Checks performed when output is configured
90  if(output->total_size() != 0)
91  {
92  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
93  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(input, output);
94  }
95 
96  return Status{};
97 }
