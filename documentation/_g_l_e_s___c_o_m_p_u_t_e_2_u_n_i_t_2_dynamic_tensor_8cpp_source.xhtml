62 /** Tests the memory manager with dynamic input and output tensors.
63  *
64  * Create and manage the tensors needed to run a simple function. After the function is executed,
65  * change the input and output size requesting more memory and go through the manage/allocate process.
66  * The memory manager should be able to update the inner structures and allocate the requested memory
67  * */
68 FIXTURE_DATA_TEST_CASE(DynamicTensorType3Single, GCDynamicTensorType3SingleFunction, framework::DatasetMode::ALL,
69  framework::dataset::zip(framework::dataset::make("Level0Shape", { TensorShape(12U, 11U, 3U), TensorShape(256U, 8U, 12U) }),
70  framework::dataset::make("Level1Shape", { TensorShape(67U, 31U, 15U), TensorShape(11U, 2U, 3U) })))
71 {
72  ARM_COMPUTE_EXPECT(internal_l0.size() == internal_l1.size(), framework::LogLevel::ERRORS);
73  ARM_COMPUTE_EXPECT(cross_l0.size() == cross_l1.size(), framework::LogLevel::ERRORS);
74 
75  const unsigned int internal_size = internal_l0.size();
76  const unsigned int cross_size = cross_l0.size();
77  if(input_l0.total_size() < input_l1.total_size())
78  {
79  for(unsigned int i = 0; i < internal_size; ++i)
80  {
81  ARM_COMPUTE_EXPECT(internal_l0[i].size < internal_l1[i].size, framework::LogLevel::ERRORS);
82  }
83  for(unsigned int i = 0; i < cross_size; ++i)
84  {
85  ARM_COMPUTE_EXPECT(cross_l0[i].size < cross_l1[i].size, framework::LogLevel::ERRORS);

95  {
