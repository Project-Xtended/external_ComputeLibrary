47 constexpr unsigned int num_elems_processed_per_iteration = 8;

49 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input1, ITensorInfo *input2, ITensorInfo *output)
50 {
51  // The window needs to be based on the output
52  Window win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
53  AccessWindowStatic input1_access(input1, 0, 0, ceil_to_multiple(input1->dimension(0), num_elems_processed_per_iteration), input1->dimension(1));
54  const unsigned int input2_right_padding = (output->dimension(0) / num_elems_processed_per_iteration) * num_elems_processed_per_iteration - input1->dimension(
55  0) + num_elems_processed_per_iteration - input2->dimension(0);
56  AccessWindowStatic input2_access(input2, -(input1->dimension(0) % num_elems_processed_per_iteration),
57  0, input2->dimension(0) + input2_right_padding, input2->dimension(1));
58  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
59  bool window_changed = update_window_and_padding(win, input1_access, input2_access, output_access);
60 
61  Window win_collapsed = win.collapse(win, Window::DimZ);
62 
63  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
64  return std::make_pair(err, win_collapsed);
65 }
66 Status validate_arguments(const ITensorInfo *input1, const ITensorInfo *input2, const ITensorInfo *output)
67 {
68  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input1, input2, output);
69  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input1);
70  ARM_COMPUTE_RETURN_ERROR_ON(input1->data_type() == DataType::UNKNOWN);
71  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input1, input2, output);
72  ARM_COMPUTE_RETURN_ERROR_ON(input1->dimension(0) + input2->dimension(0) > output->dimension(0));
73 
74  for(size_t i = 1; i < Coordinates::num_max_dimensions; ++i)
75  {
76  ARM_COMPUTE_RETURN_ERROR_ON(input1->dimension(i) != output->dimension(i));
77  ARM_COMPUTE_RETURN_ERROR_ON(input2->dimension(i) != output->dimension(i));
78  }
79  ARM_COMPUTE_RETURN_ERROR_ON(input1->num_dimensions() > 4);
80 
81  return Status{};
82 }

97 void CLWidthConcatenate2TensorsKernel::configure(const ICLTensor *input1, const ICLTensor *input2, ICLTensor *output)
98 {
99  ARM_COMPUTE_ERROR_ON_NULLPTR(input1, input2, output);
100  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input1->info(), input2->info(), output->info()));
101 
102  _input1 = input1;
103  _input2 = input2;
104  _output = output;
105 
106  // Add build options
107  CLBuildOptions build_opts;
108  build_opts.add_option("-DDATA_TYPE=" + get_underlying_cl_type_from_data_type(input1->info()->data_type()));
109  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(num_elems_processed_per_iteration));
110  build_opts.add_option("-DDEPTH=" + support::cpp11::to_string(input1->info()->dimension(2)));
111  build_opts.add_option("-DINPUT1_WIDTH=" + support::cpp11::to_string(input1->info()->dimension(0)));
112  build_opts.add_option("-DELEMENT_SIZE=" + support::cpp11::to_string(input1->info()->element_size()));
113 
114  // If input have different quantization info set quantization parameters needed for the re-quantization process
115  const bool have_different_qinfo = helpers::tensor_info::tensors_have_different_quantization_info(output->info(), input1->info(), input2->info());
116  if(is_data_type_quantized_asymmetric(input1->info()->data_type()) && have_different_qinfo)
117  {
118  const UniformQuantizationInfo iq1_info = input1->info()->quantization_info().uniform();
119  const UniformQuantizationInfo iq2_info = input2->info()->quantization_info().uniform();
120  const UniformQuantizationInfo oq_info = output->info()->quantization_info().uniform();
121 
122  build_opts.add_option("-DOFFSET_IN1=" + float_to_string_with_full_precision(iq1_info.offset));
123  build_opts.add_option("-DSCALE_IN1=" + float_to_string_with_full_precision(iq1_info.scale));
124  build_opts.add_option("-DOFFSET_IN2=" + float_to_string_with_full_precision(iq2_info.offset));
125  build_opts.add_option("-DSCALE_IN2=" + float_to_string_with_full_precision(iq2_info.scale));
126  build_opts.add_option("-DOFFSET_OUT=" + float_to_string_with_full_precision(oq_info.offset));
127  build_opts.add_option("-DSCALE_OUT=" + float_to_string_with_full_precision(oq_info.scale));
128  }
129 
130  // Create kernel
131  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("concatenate_width_x2", build_opts.options()));
132 
133  // Configure kernel window
134  auto win_config = validate_and_configure_window(input1->info(), input2->info(), output->info());
135  ARM_COMPUTE_ERROR_THROW_ON(std::get<0>(win_config));
136 
137  ICLKernel::configure_internal(std::get<1>(win_config));
138 
139  // Set output valid region
140  output->info()->set_valid_region(ValidRegion(Coordinates(), output->info()->tensor_shape()));
141 
142  // Pass paddings as arguments to the kernel
143  const unsigned int input1_width = input1->info()->dimension(0);
144  const unsigned int input1_right_padding = ceil_to_multiple(input1_width, num_elems_processed_per_iteration) - input1_width;
145  const unsigned int input2_left_padding = input1_width % num_elems_processed_per_iteration;
146  unsigned int idx0 = 3 * num_arguments_per_4D_tensor();
147  _kernel.setArg<cl_uint>(idx0++, input1_right_padding);
148  _kernel.setArg<cl_uint>(idx0++, input2_left_padding);
149 
150  // Set config_id for enabling LWS tuning
151  _config_id = "concatenate_width_x2_";
152  _config_id += lower_string(string_from_data_type(input1->info()->data_type()));
153  _config_id += "_";
154  _config_id += support::cpp11::to_string(input1->info()->dimension(0));
155  _config_id += "_";
156  _config_id += support::cpp11::to_string(input1->info()->dimension(1));
157  _config_id += "_";
158  _config_id += support::cpp11::to_string(input2->info()->dimension(0));
159  _config_id += "_";
160  _config_id += support::cpp11::to_string(input2->info()->dimension(1));
161 }

163 void CLWidthConcatenate2TensorsKernel::run(const Window &window, cl::CommandQueue &queue)
164 {
165  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
166  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
167 
168  Window slice = window.first_slice_window_4D();
169 
170  do
171  {
172  unsigned int idx = 0;
173  add_4D_tensor_argument(idx, _input1, slice);
174  add_4D_tensor_argument(idx, _input2, slice);
175  add_4D_tensor_argument(idx, _output, slice);
176  enqueue(queue, *this, window, lws_hint());
177  }
178  while(window.slide_window_slice_4D(slice));
179 }
