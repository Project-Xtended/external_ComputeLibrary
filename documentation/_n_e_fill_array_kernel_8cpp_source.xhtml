39 void NEFillArrayKernel::configure(const IImage *input, uint8_t threshold, IKeyPointArray *output)
40 {
41  ARM_COMPUTE_ERROR_ON_TENSOR_NOT_2D(input);
42  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::U8);
43  ARM_COMPUTE_ERROR_ON(nullptr == output);
44 
45  _input = input;
46  _output = output;
47  _threshold = threshold;

49  constexpr unsigned int num_elems_processed_per_iteration = 1;
50  constexpr unsigned int num_elems_read_per_iteration = 1;
51 
52  // Configure kernel window

65 void NEFillArrayKernel::run(const Window &window, const ThreadInfo &info)
66 {
67  ARM_COMPUTE_UNUSED(info);
68  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
69  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(INEKernel::window(), window);
70 
71  Iterator input(_input, window);
72 
73  execute_window_loop(window, [&](const Coordinates & id)
74  {
75  const uint8_t value = *input.ptr();
76 
77  if(value >= _threshold)
78  {
79  KeyPoint p;
80  p.x = id.x();
81  p.y = id.y();
82  p.strength = value;
83  p.tracking_status = 1;
84  p.scale = 0.f;
85  p.orientation = 0.f;
86  p.error = 0.f;
87 
88  if(!_output->push_back(p))
89  {
90  return; //Overflowed: stop trying to add more points
91  }
92  }
93  },
94  input);
95 }
