143 #define CAST_SUITE(NAME, idt, odt, type, dataset, tolerance) \
144  TEST_SUITE(NAME) \
145  DATA_TEST_CASE(Configuration, framework::DatasetMode::ALL, combine(datasets::SmallShapes(), datasets::ConvertPolicies()), \
146  shape, policy) \
147  { \
148  CLTensor src = create_tensor<CLTensor>(shape, idt, 1); \
149  CLTensor dst = create_tensor<CLTensor>(shape, odt, 1); \
150  \
151  CLCast cast; \
152  cast.configure(&src, &dst, policy); \
153  \
154  const ValidRegion valid_region = shape_to_valid_region(shape); \
155  validate(dst.info()->valid_region(), valid_region); \
156  \
157  const PaddingSize padding = PaddingCalculator(shape.x(), 16).required_padding(); \
158  validate(src.info()->padding(), padding); \
159  validate(dst.info()->padding(), padding); \
160  } \
161  FIXTURE_DATA_TEST_CASE(RunSmall, type, framework::DatasetMode::PRECOMMIT, combine(combine(datasets::SmallShapes(), dataset), \
162  datasets::ConvertPolicies())) \
163  { \
164  validate(CLAccessor(_target), _reference, tolerance); \
165  } \
166  TEST_SUITE_END()
