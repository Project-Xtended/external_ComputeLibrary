58 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const PoolingLayerInfo &pool_info)
59 {
60  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
61  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
62  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::F16, DataType::F32);
63  ARM_COMPUTE_RETURN_ERROR_ON_MSG((is_data_type_quantized_asymmetric(input->data_type()) && pool_info.pool_type == PoolingType::L2),
64  "Unsupported combination of parameters!");
65  ARM_COMPUTE_RETURN_ERROR_ON_MSG(is_data_type_quantized(input->data_type()) && !pool_info.exclude_padding && (pool_info.pool_type == PoolingType::AVG) && pool_info.pad_stride_info.has_padding()
66  && (input->data_layout() == DataLayout::NHWC),
67  "exclude_padding equal false is not supported for AVG Pooling with padding on quantized types");
68 
69  // Checks performed when output is configured
70  if(output->total_size() != 0)
71  {
72  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
73  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_LAYOUT(input, output);
74  TensorInfo out_info(TensorInfo(compute_pool_shape(*input, pool_info), 1, output->data_type()));
75  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_SHAPES(output, &out_info);
76  }
77 
78  return Status{};

81 std::tuple<Status, Window, CLPoolingConfig> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output, const PoolingLayerInfo &pool_info)
82 {
83  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
84 
85  // Get data layout
86  const DataLayout data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? input->data_layout() : pool_info.data_layout;
87  const int idx_width = get_data_layout_dimension_index(data_layout, DataLayoutDimension::WIDTH);
88  const int idx_height = get_data_layout_dimension_index(data_layout, DataLayoutDimension::HEIGHT);
89 
90  int pool_stride_x = 0;
91  int pool_stride_y = 0;
92  unsigned int pooled_w = 0;
93  unsigned int pooled_h = 0;
94  int pool_size_x = pool_info.is_global_pooling ? input->dimension(idx_width) : pool_info.pool_size.width;
95  int pool_size_y = pool_info.is_global_pooling ? input->dimension(idx_height) : pool_info.pool_size.height;
96  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
97  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
98  const int pool_pad_right = pad_stride_info.pad_right();
99  const int pool_pad_top = pad_stride_info.pad_top();
100  const int pool_pad_left = pad_stride_info.pad_left();
101  const int pool_pad_bottom = pad_stride_info.pad_bottom();
102  BorderSize border_size = BorderSize(pool_pad_top, pool_pad_right, pool_pad_bottom, pool_pad_left);
103 
104  auto_init(input, output, pool_info);
105  pooled_w = output->tensor_shape()[idx_width];
106  pooled_h = output->tensor_shape()[idx_height];
107 
108  const DataType data_type = input->data_type();
109 
110  const int input_width = input->dimension(idx_width);
111  const int input_height = input->dimension(idx_height);
112 
113  unsigned int num_elems_processed_per_iteration = 0;
114  bool window_changed = false;
115  Window win{};
116  switch(data_layout)
117  {
118  case DataLayout::NCHW:
119  {
120  // Change the number of elements processed per iteration
121  // for pooling 3x3 with stride less equal than 3
122  const bool can_optimize = (pool_size_x == 3) && (pool_size_y == 3) && (pool_stride_x <= 3) && !is_data_type_quantized(data_type);
123  num_elems_processed_per_iteration = can_optimize ? 4 : 1;
124  const unsigned int num_elems_read_per_iteration = (num_elems_processed_per_iteration - 1) * pool_stride_x + pool_size_x;
125 
126  // Number of iterations in X dimension
127  const int num_iterations_x = (pooled_w + num_elems_processed_per_iteration - 1) / num_elems_processed_per_iteration;
128 
129  // Upper limit for the number of right/bottom border elements that are accessed
130  const int upper_bound_w = ((num_iterations_x - 1) * num_elems_processed_per_iteration * pool_stride_x - pool_pad_left + num_elems_read_per_iteration) - input_width;
131  const int upper_bound_h = ((pooled_h - 1) * pool_stride_y - pool_pad_top + pool_size_y) - input_height;
132 
133  border_size.right = std::max(upper_bound_w, pool_pad_right);
134  border_size.bottom = std::max(upper_bound_h, pool_pad_bottom);
135 
136  win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
137 
138  AccessWindowRectangle input_access(input, -pool_pad_left, -pool_pad_top, num_elems_read_per_iteration, pool_size_y,
139  pool_stride_x, pool_stride_y);
140  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
141  window_changed = update_window_and_padding(win, input_access, output_access);
142  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
143  break;
144  }
145  case DataLayout::NHWC:
146  {
147  num_elems_processed_per_iteration = 8;
148  win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
149 
150  AccessWindowStatic input_access(input,
151  0, -1,
152  ceil_to_multiple(input->dimension(0), num_elems_processed_per_iteration), input->dimension(1));
153  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
154  window_changed = update_window_and_padding(win, input_access, output_access);
155  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
156  break;
157  }
158  default:
159  ARM_COMPUTE_ERROR("Not implemented");
160  }
161 
162  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
163  return std::make_tuple(err, win, CLPoolingConfig(num_elems_processed_per_iteration, border_size));
164 }

177 void CLPoolingLayerKernel::configure(const ICLTensor *input, ICLTensor *output, const PoolingLayerInfo &pool_info)
178 {
179  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
180 
181  // Set instance variables
182  _input = input;
183  _output = output;
184  _pool_info = pool_info;
185  _data_layout = pool_info.data_layout == DataLayout::UNKNOWN ? input->info()->data_layout() : pool_info.data_layout;
186 
187  int pool_stride_x = 0;
188  int pool_stride_y = 0;
189  const PoolingType pool_type = pool_info.pool_type;
190  const int idx_width = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::WIDTH);
191  const int idx_height = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::HEIGHT);
192  const int idx_channel = get_data_layout_dimension_index(_data_layout, DataLayoutDimension::CHANNEL);
193  const int pool_size_x = pool_info.is_global_pooling ? input->info()->dimension(idx_width) : pool_info.pool_size.width;
194  const int pool_size_y = pool_info.is_global_pooling ? input->info()->dimension(idx_height) : pool_info.pool_size.height;
195  const PadStrideInfo pad_stride_info = pool_info.pad_stride_info;
196  const bool exclude_padding = pool_info.exclude_padding;
197  std::tie(pool_stride_x, pool_stride_y) = pad_stride_info.stride();
198  const int pool_pad_top = pad_stride_info.pad_top();
199  const int pool_pad_left = pad_stride_info.pad_left();
200 
201  // Set build options
202  CLBuildOptions build_opts;
203 
204  if(is_data_type_quantized_asymmetric(input->info()->data_type()) && input->info()->quantization_info() != output->info()->quantization_info())
205  {
206  const UniformQuantizationInfo iq_info = input->info()->quantization_info().uniform();
207  const UniformQuantizationInfo oq_info = output->info()->quantization_info().uniform();
208 
209  build_opts.add_option("-DOFFSET_IN1=" + float_to_string_with_full_precision(iq_info.offset));
210  build_opts.add_option("-DOFFSET_OUT=" + float_to_string_with_full_precision(oq_info.offset));
211  build_opts.add_option("-DSCALE_IN1=" + float_to_string_with_full_precision(iq_info.scale));
212  build_opts.add_option("-DSCALE_OUT=" + float_to_string_with_full_precision(oq_info.scale));
213  }
214 
215  // Check output dimensions
216  auto_init(input->info(), output->info(), pool_info);
217  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), pool_info));
218 
219  const DataType data_type = input->info()->data_type();
220 
221  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
222  build_opts.add_option("-DPOOL_" + string_from_pooling_type(pool_type));
223  build_opts.add_option("-DSTRIDE_X=" + support::cpp11::to_string(pool_stride_x));
224  build_opts.add_option("-DSTRIDE_Y=" + support::cpp11::to_string(pool_stride_y));
225  build_opts.add_option("-DPAD_X=" + support::cpp11::to_string(pool_pad_left));
226  build_opts.add_option("-DPAD_Y=" + support::cpp11::to_string(pool_pad_top));
227  build_opts.add_option("-DPOOL_SIZE_X=" + support::cpp11::to_string(pool_size_x));
228  build_opts.add_option("-DPOOL_SIZE_Y=" + support::cpp11::to_string(pool_size_y));
229 
230  // Set the initial value for the pooling operation accordingly with the data type
231  if(pool_type == PoolingType::MAX)
232  {
233  if(is_data_type_quantized(data_type))
234  {
235  PixelValue type_min{};
236  std::tie(type_min, std::ignore) = get_min_max(data_type);
237  build_opts.add_option("-DINITIAL_VALUE=" + support::cpp11::to_string(type_min.get<int32_t>()));
238  }
239  else
240  {
241  build_opts.add_option("-DINITIAL_VALUE=" + float_to_string_with_full_precision(std::numeric_limits<float>::lowest()));
242  }
243  }
244  else
245  {
246  // Pool AVG and Pool L2 initial value
247  build_opts.add_option("-DINITIAL_VALUE=0");
248  }
249 
250  const auto use_fp_mixed_precision = (data_type == DataType::F16) && pool_info.fp_mixed_precision;
251  const auto use_wider_accumulator = use_fp_mixed_precision && (pool_type != PoolingType::MAX);
252  const auto acc_data_type = get_cl_type_from_data_type(use_wider_accumulator ? DataType::F32 : data_type);
253  build_opts.add_option("-DACC_DATA_TYPE=" + acc_data_type);
254  build_opts.add_option_if(use_wider_accumulator, "-DFP_MIXED_PRECISION");
255 
256  // Create kernel
257  switch(_data_layout)
258  {
259  case DataLayout::NCHW:
260  {
261  build_opts.add_option("-DMAX_WIDTH=" + support::cpp11::to_string(input->info()->dimension(idx_width) + (exclude_padding ? 0 : pool_pad_left)));
262  build_opts.add_option("-DMAX_HEIGHT=" + support::cpp11::to_string(input->info()->dimension(idx_height) + (exclude_padding ? 0 : pool_pad_top)));
263  if(pool_type != PoolingType::MAX)
264  {
265  build_opts.add_option_if(exclude_padding, "-DEXCLUDE_PADDING");
266  }
267 
268  if((pool_size_x == 3) && (pool_size_y == 3) && !is_data_type_quantized_asymmetric(data_type))
269  {
270  // Check if we have pool3x3 with stride_x less equal than 3. In these cases, run an optimized OpenCL kernel where
271  // each thread computes 4 output elements
272  const bool is_pool3x3_stride_le3 = (pool_size_x == 3) && (pool_size_y == 3) && (pool_stride_x <= 3);
273 
274  std::string kernel_name = ((is_pool3x3_stride_le3) ? "pooling_layer_optimized_" : "pooling_layer_")
275  + support::cpp11::to_string(pool_size_x);
276  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
277  }
278  else // Run general case
279  {
280  std::string kernel_name = is_data_type_quantized_asymmetric(data_type) ? "pooling_layer_MxN_quantized_nchw" : "pooling_layer_MxN_nchw";
281  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
282  }
283  break;
284  }
285  case DataLayout::NHWC:
286  {
287  build_opts.add_option_if(exclude_padding, "-DEXCLUDE_PADDING");
288  build_opts.add_option("-DMAX_WIDTH=" + support::cpp11::to_string(input->info()->dimension(idx_width)));
289  build_opts.add_option("-DMAX_HEIGHT=" + support::cpp11::to_string(input->info()->dimension(idx_height)));
290  build_opts.add_option_if(output->info()->tensor_shape().total_size_upper(3) > 1,
291  "-DDST_DEPTH=" + support::cpp11::to_string(output->info()->dimension(idx_height)));
292  std::string kernel_name = is_data_type_quantized_asymmetric(data_type) ? "pooling_layer_MxN_quantized_nhwc" : "pooling_layer_MxN_nhwc";
293  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
294  break;
295  }
296  default:
297  ARM_COMPUTE_ERROR("Not implemented");
298  }
299 
300  // Configure kernel window
301  auto win_config = validate_and_configure_window(input->info(), output->info(), pool_info);
302 
303  ARM_COMPUTE_ERROR_THROW_ON(std::get<0>(win_config));
304  ICLKernel::configure_internal(std::get<1>(win_config));
305 
306  if(_data_layout == DataLayout::NCHW)
307  {
308  CLPoolingConfig pooling_config = std::get<2>(win_config);
309  _num_elems_processed_per_iteration = pooling_config.first;
310  _border_size = pooling_config.second;
311  }
312  else
313  {
314  _border_size = BorderSize(1, 0, 0, 0);
315  _num_elems_processed_per_iteration = 8;
316  }
317 
318  // Set config_id for enabling LWS tuning
319  _config_id = "pooling_layer_";
320  _config_id += lower_string(string_from_data_type(data_type));
321  _config_id += "_";
322  _config_id += lower_string(string_from_data_layout(_data_layout));
323  _config_id += "_";
324  _config_id += support::cpp11::to_string(output->info()->dimension(idx_width));
325  _config_id += "_";
326  _config_id += support::cpp11::to_string(output->info()->dimension(idx_height));
327  _config_id += "_";
328  _config_id += support::cpp11::to_string(output->info()->dimension(idx_channel));
329  _config_id += "_";
330  _config_id += lower_string(string_from_data_layout(input->info()->data_layout()));
331 }

341 void CLPoolingLayerKernel::run(const Window &window, cl::CommandQueue &queue)
342 {
343  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
344  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
345 
346  unsigned int pool_stride_x = 0;
347  unsigned int pool_stride_y = 0;
348  std::tie(pool_stride_x, pool_stride_y) = _pool_info.pad_stride_info.stride();
349 
350  // Collapse window
351  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
352 
353  switch(_data_layout)
354  {
355  case DataLayout::NCHW:
356  {
357  Window slice = window_collapsed.first_slice_window_3D();
358  do
359  {
360  // Upsample input by pool size
361  Window in_slice(slice);
362  in_slice.set(Window::DimX, Window::Dimension(in_slice.x().start() - _pool_info.pad_stride_info.pad_left(),
363  (in_slice.x().end() - _pool_info.pad_stride_info.pad_left()) * pool_stride_x,
364  pool_stride_x * _num_elems_processed_per_iteration));
365  in_slice.set(Window::DimY, Window::Dimension(in_slice.y().start() - _pool_info.pad_stride_info.pad_top(),
366  (in_slice.y().end() - _pool_info.pad_stride_info.pad_top()) * pool_stride_y,
367  pool_stride_y));
368 
369  // Set inputs
370  unsigned int idx = 0;
371  add_3D_tensor_argument(idx, _input, in_slice);
372  add_3D_tensor_argument(idx, _output, slice);
373  enqueue(queue, *this, slice, lws_hint());
374  }
375  while(window_collapsed.slide_window_slice_3D(slice));
376  break;
377  }
378  case DataLayout::NHWC:
379  {
380  const size_t total_batches = _output->info()->tensor_shape().total_size_upper(3);
381 
382  Window slice = window_collapsed.first_slice_window_4D();
383  Window in_slice = window_collapsed.first_slice_window_4D();
384  in_slice.set(Window::DimX, Window::Dimension(0, _input->info()->dimension(0), _num_elems_processed_per_iteration));
385  in_slice.set(Window::DimY, Window::Dimension(0, _input->info()->dimension(1), pool_stride_x));
386  in_slice.set(Window::DimZ, Window::Dimension(0, _input->info()->dimension(2), pool_stride_y));
387  in_slice.set(3, Window::Dimension(0, total_batches, 1));
388  do
389  {
390  // Set inputs
391  unsigned int idx = 0;
392  add_4D_tensor_argument(idx, _input, in_slice);
393  add_4D_tensor_argument(idx, _output, slice);
394  enqueue(queue, *this, slice, lws_hint());
395  }
396  while(window.slide_window_slice_4D(slice) && window.slide_window_slice_4D(in_slice));
397  break;
398  }
399  default:
400  ARM_COMPUTE_ERROR("Not implemented");
401  }
