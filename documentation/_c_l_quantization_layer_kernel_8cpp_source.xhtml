55 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output)
56 {
57  // Configure kernel window
58  Window win = calculate_max_window(*input, Steps());
59 
60  const int vec_size_x = 16 / input->element_size();
61  const int input_width_x = input->tensor_shape().x();
62  const bool multi_access_x = (input_width_x / vec_size_x > 0);
63  if(multi_access_x)
64  {
65  win.set(Window::DimX, Window::Dimension(win.x().start(), ceil_to_multiple(win.x().end(), vec_size_x), vec_size_x));
66  }
67 
68  Coordinates coord;
69  coord.set_num_dimensions(output->num_dimensions());
70  output->set_valid_region(ValidRegion(coord, output->tensor_shape()));
71 
72  return std::make_pair(Status{}, win);
73 }

81 void CLQuantizationLayerKernel::configure(const ICLTensor *input, ICLTensor *output)
82 {
83  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
84  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info()));
85 
86  _input = input;
87  _output = output;
88 
89  const int vec_size_x = 16 / input->info()->element_size();
90  const int input_width_x = input->info()->tensor_shape().x();
91  const bool multi_access_x = (input_width_x / vec_size_x > 0);
92 
93  // Configure kernel window
94  auto win_config = validate_and_configure_window(input->info(), output->info());
95  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
96  ICLKernel::configure_internal(win_config.second);
97 
98  const UniformQuantizationInfo qinfo = output->info()->quantization_info().uniform();
99  const DataType output_data_type = output->info()->data_type();
100 
101  float scale_to_apply = qinfo.scale;
102  int32_t offset_to_apply = qinfo.offset;
103  if(is_data_type_quantized_asymmetric(_input->info()->data_type()))
104  {
105  /*
106  * In case of requantization of a quantized input tensor to an output tensor with another quantization
107  * instead of of apply dequantization and then a quantization functions, we just compute new scale and
108  * offset to apply.
109  *
110  * Assuming:
111  * - q_i as input quantized value
112  * - q_o as output quantized value
113  * - z_i as input quantization offset value
114  * - z_o as output quantization offset value
115  * - s_i as input quantization scale value
116  * - s_o as output quantization scale value
117  * - z_n as new quantization offset value
118  * - s_n as new quantization scale value
119  *
120  * q_o = ( q_i - z_i ) * s_i / s_o + z_o
121  *
122  * We can rewrite the formula as:
123  *
124  * q_o = ( q_i * s_i / s_o ) - z_i * s_i / s_o + z_o
125  *
126  * q_o = q_i / s_n + z_n
127  *
128  * Where:
129  *
130  * s_n = s_o / s_i
131  *
132  * z_n = - z_i * s_i / s_o + z_o
133  *
134  */
135  const UniformQuantizationInfo qinfo_in = _input->info()->quantization_info().uniform();
136  scale_to_apply /= qinfo_in.scale;
137  // In order to minimize flooring we convert the offset to a float,
138  // then compute the new offset in the float domain,
139  // finally we convert it back as int32_t
140  offset_to_apply -= static_cast<int32_t>(static_cast<float>(qinfo_in.offset) * qinfo_in.scale / qinfo.scale);
141  }
142 
143  // Create kernel
144  CLBuildOptions build_opts;
145  build_opts.add_option_if(is_data_type_float(_input->info()->data_type()), "-DIS_FLOAT");
146  build_opts.add_option("-DSCALE=" + float_to_string_with_full_precision(scale_to_apply));
147  build_opts.add_option("-DOFFSET=" + support::cpp11::to_string(offset_to_apply));
148  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(vec_size_x));
149  build_opts.add_option("-DDATA_TYPE_IN=" + get_cl_type_from_data_type(input->info()->data_type()));
150  build_opts.add_option("-DDATA_TYPE_OUT=" + get_cl_type_from_data_type(output_data_type));
151  build_opts.add_option_if(multi_access_x, "-DLAST_ACCESSED_X=" + support::cpp11::to_string(std::max<int>(input_width_x - vec_size_x, 0)));
152  std::pair<int, int> min_max_quant_values = quantization::get_min_max_values_from_quantized_data_type(output_data_type);
153  build_opts.add_option("-DMIN_QUANT_VAL=" + support::cpp11::to_string(min_max_quant_values.first));
154  build_opts.add_option("-DMAX_QUANT_VAL=" + support::cpp11::to_string(min_max_quant_values.second));
155 
156  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("quantization_layer", build_opts.options()));
157 }

167 void CLQuantizationLayerKernel::run(const Window &window, cl::CommandQueue &queue)
168 {
169  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
170  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
171 
172  Window window_collapsed = window.collapse_if_possible(ICLKernel::window(), 3);
173  Window slice = window_collapsed.first_slice_window_3D();
174 
175  do
176  {
177  unsigned int idx = 0;
178  add_3D_tensor_argument(idx, _input, slice);
179  add_3D_tensor_argument(idx, _output, slice);
180  enqueue(queue, *this, slice, lws_hint());
181  }
182  while(window_collapsed.slide_window_slice_3D(slice));
183 }
