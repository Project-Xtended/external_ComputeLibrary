46 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output,
47  const PadStrideInfo &conv_info, unsigned int depth_multiplier, const ActivationLayerInfo &act_info, const Size2D dilation,
48  const ITensorInfo *output_multipliers, const ITensorInfo *output_shifts)
49 {
50  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
51  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::QASYMM8, DataType::QASYMM8_SIGNED, DataType::F16, DataType::F32);
52  ARM_COMPUTE_RETURN_ERROR_ON_MSG((act_info.enabled()) && (input->data_type() == DataType::QASYMM8 || input->data_type() == DataType::QASYMM8_SIGNED)
53  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::LU_BOUNDED_RELU)
54  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::BOUNDED_RELU)
55  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::RELU)
56  && (act_info.activation() != ActivationLayerInfo::ActivationFunction::LOGISTIC),
57  "For QASYMM8 only logistic, relu, lower bounded relu and lower-upper bounded relu are supported");
58  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(0) != 3 || weights->dimension(1) != 3);
59  ARM_COMPUTE_RETURN_ERROR_ON(conv_info.stride().first < 1 || conv_info.stride().first > 3);
60 
61  ARM_COMPUTE_RETURN_ERROR_ON((dilation.x() < 1) || (dilation.y() < 1));
62 
63  const bool is_qasymm = is_data_type_quantized_asymmetric(input->data_type());
64 
65  if(biases != nullptr)
66  {
67  if(is_qasymm)
68  {
69  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(biases, 1, DataType::S32);
70  }
71  else
72  {
73  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(weights, biases);
74  }
75  ARM_COMPUTE_RETURN_ERROR_ON((biases->dimension(0) != weights->dimension(2)) && (weights->dimension(2) != 1 || biases->dimension(0) != weights->dimension(3)));
76  ARM_COMPUTE_RETURN_ERROR_ON(biases->num_dimensions() > 1);
77  }
78 
79  if(is_qasymm)
80  {
81  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(output_multipliers, output_shifts);
82  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output_multipliers, 1, DataType::S32);
83  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output_shifts, 1, DataType::S32);
84  ARM_COMPUTE_RETURN_ERROR_ON(output_multipliers->num_dimensions() > 1);
85  ARM_COMPUTE_RETURN_ERROR_ON(output_shifts->num_dimensions() > 1);
86 
87  if(is_data_type_quantized_per_channel(weights->data_type()))
88  {
89  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(weights, 1, DataType::QSYMM8_PER_CHANNEL);
90  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(2) != output_multipliers->dimension(0));
91  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(2) != output_shifts->dimension(0));
92  }
93  else
94  {
95  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
96  ARM_COMPUTE_RETURN_ERROR_ON(1 != output_multipliers->dimension(0));
97  ARM_COMPUTE_RETURN_ERROR_ON(1 != output_shifts->dimension(0));
98  }
99  }
100  else
101  {
102  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
103  }
104 
105  if(output->total_size() != 0)
106  {
107  const TensorShape output_shape = compute_depthwise_convolution_shape(*input, *weights, conv_info, depth_multiplier, dilation);
108  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), output_shape);
109  }
110 
111  return Status{};
112 }

114 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *weights, ITensorInfo *output, const PadStrideInfo &conv_info,
115  unsigned int depth_multiplier, GPUTarget gpu_target, std::string &kernel_name, const Size2D dilation)
116 {
117  // Output auto inizialitation if not yet initialized
118  const TensorShape output_shape = compute_depthwise_convolution_shape(*input, *weights, conv_info, depth_multiplier, dilation);
119  auto_init_if_empty(*output, input->clone()->set_tensor_shape(output_shape).set_quantization_info(output->quantization_info()));
120 
121  const unsigned int conv_stride_x = conv_info.stride().first;
122  const unsigned int conv_stride_y = conv_info.stride().second;
123  const bool is_qasymm = is_data_type_quantized_asymmetric(input->data_type());
124  const bool is_bifrost = get_arch_from_target(gpu_target) == GPUTarget::BIFROST;
125 
126  // Configure kernel window
127  unsigned int num_elems_read_per_iteration_x = 0;
128  unsigned int num_elems_read_per_iteration_y = 0;
129  unsigned int num_elems_written_per_iteration_x = 0;
130  unsigned int num_elems_written_per_iteration_y = 0;
131 
132  if(input->data_type() == DataType::F16)
133  {
134  kernel_name = "depthwise_convolution_3x3_f16";
135  num_elems_written_per_iteration_x = 8 / data_size_from_type(input->data_type());
136  num_elems_written_per_iteration_y = 1;
137  num_elems_read_per_iteration_y = 3;
138  switch(conv_stride_x)
139  {
140  case 1:
141  num_elems_read_per_iteration_x = 8;
142  break;
143  case 2:
144  num_elems_read_per_iteration_x = 9;
145  break;
146  case 3:
147  num_elems_read_per_iteration_x = 16;
148  break;
149  default:
150  num_elems_read_per_iteration_x = 3 + (num_elems_written_per_iteration_x - 1) * conv_stride_x;
151  break;
152  }
153  if(is_bifrost)
154  {
155  if(conv_stride_x == 1 && conv_stride_y == 1)
156  {
157  kernel_name = "depthwise_convolution_3x3_stridex1_stridey1_bifrost_f16";
158  num_elems_read_per_iteration_x = 8;
159  num_elems_written_per_iteration_x = 4;
160  num_elems_read_per_iteration_y = 6;
161  num_elems_written_per_iteration_y = 4;
162  }
163  else if(conv_stride_x == 2 && conv_stride_y == 2)
164  {
165  kernel_name = "depthwise_convolution_3x3_stridex2_stridey2_bifrost_f16";
166  num_elems_read_per_iteration_x = 10;
167  num_elems_written_per_iteration_x = 4;
168  num_elems_read_per_iteration_y = 5;
169  num_elems_written_per_iteration_y = 2;
170  }
171  }
172  }
173  else if(input->data_type() == DataType::F32 && is_bifrost)
174  {
175  if(conv_stride_x == 1 && conv_stride_y == 1)
176  {
177  kernel_name = "depthwise_convolution_3x3_stridex1_stridey1_bifrost_f32";
178  num_elems_read_per_iteration_x = 4;
179  num_elems_read_per_iteration_y = 6;
180  num_elems_written_per_iteration_x = 2;
181  num_elems_written_per_iteration_y = 4;
182  }
183  else if(conv_stride_x == 2 && conv_stride_y == 2)
184  {
185  kernel_name = "depthwise_convolution_3x3_stridex2_stridey2_bifrost_f32";
186  num_elems_read_per_iteration_x = 6;
187  num_elems_read_per_iteration_y = 5;
188  num_elems_written_per_iteration_x = 2;
189  num_elems_written_per_iteration_y = 2;
190  }
191  else
192  {
193  kernel_name = "depthwise_convolution_3x3";
194  num_elems_written_per_iteration_x = 8 / data_size_from_type(input->data_type());
195  num_elems_written_per_iteration_y = 1;
196  num_elems_read_per_iteration_x = 3 + (num_elems_written_per_iteration_x - 1) * conv_stride_x;
197  num_elems_read_per_iteration_y = 3;
198  }
199  }
200  else
201  {
202  const bool is_dot8_supported = dot8_supported(CLKernelLibrary::get().get_device()) && !is_data_type_quantized_per_channel(weights->data_type());
203 
204  kernel_name = is_qasymm ? "dwc_3x3_native_quantized8" : "depthwise_convolution_3x3";
205  kernel_name += (is_qasymm && is_dot8_supported ? "_dot8" : "");
206  kernel_name += (is_qasymm ? "_nchw" : "");
207 
208  num_elems_written_per_iteration_x = 8 / data_size_from_type(input->data_type());
209  num_elems_written_per_iteration_y = (is_qasymm && conv_stride_y == 1 && dilation.y() == 1) ? 2 : 1;
210  num_elems_read_per_iteration_x = 3 + (num_elems_written_per_iteration_x - 1) * conv_stride_x + (conv_stride_x > 1 ? 1 : 0);
211  num_elems_read_per_iteration_y = num_elems_written_per_iteration_y + 2;
212  }
213  num_elems_read_per_iteration_x += (num_elems_read_per_iteration_x - 1) * (dilation.x() - 1);
214  num_elems_read_per_iteration_y += (num_elems_read_per_iteration_y - 1) * (dilation.y() - 1);
215 
216  // Create window and update padding
217  Window win = calculate_max_window(*output, Steps(num_elems_written_per_iteration_x, num_elems_written_per_iteration_y));
218 
219  AccessWindowRectangle input_access(input, -conv_info.pad_left(), -conv_info.pad_top(),
220  num_elems_read_per_iteration_x, num_elems_read_per_iteration_y,
221  conv_stride_x, conv_stride_y);
222  AccessWindowStatic weights_access(weights, 0, 0, 3, 3);
223  AccessWindowRectangle output_access(output, 0, 0, num_elems_written_per_iteration_x, num_elems_written_per_iteration_y);
224 
225  bool window_changed = update_window_and_padding(win, input_access, weights_access, output_access);
226 
227  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
228 
229  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
230  return std::make_pair(err, win);
231 }

244 void CLDepthwiseConvolutionLayer3x3NCHWKernel::configure(const ICLTensor *input, const ICLTensor *weights, const ICLTensor *biases, ICLTensor *output,
245  const PadStrideInfo &conv_info, unsigned int depth_multiplier, ActivationLayerInfo act_info, const Size2D &dilation,
246  const ICLTensor *output_multipliers, const ICLTensor *output_shifts)
247 {
248  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
249  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), weights->info(), (biases != nullptr) ? biases->info() : nullptr, output->info(),
250  conv_info, depth_multiplier, act_info, dilation,
251  (output_multipliers != nullptr) ? output_multipliers->info() : nullptr,
252  (output_shifts != nullptr) ? output_shifts->info() : nullptr));
253 
254  _input = input;
255  _output = output;
256  _weights = weights;
257  _biases = biases;
258  _conv_stride_x = conv_info.stride().first;
259  _conv_stride_y = conv_info.stride().second;
260  _conv_pad_left = conv_info.pad_left();
261  _conv_pad_top = conv_info.pad_top();
262  _border_size = BorderSize(_conv_pad_top, conv_info.pad_right(), conv_info.pad_bottom(), _conv_pad_left);
263  _output_multipliers = output_multipliers;
264  _output_shifts = output_shifts;
265  _is_quantized = is_data_type_quantized_asymmetric(input->info()->data_type());
266 
267  // Configure kernel window
268  std::string kernel_name;
269  const GPUTarget gpu_target = get_target();
270 
271  auto win_config = validate_and_configure_window(input->info(), weights->info(), output->info(), conv_info, depth_multiplier, gpu_target, kernel_name, dilation);
272  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
273  ICLKernel::configure_internal(win_config.second);
274 
275  // Set build options
276  CLBuildOptions build_opts;
277  build_opts.add_option("-DACTIVATION_TYPE=" + lower_string(string_from_activation_func(act_info.activation())));
278  build_opts.add_option("-DDST_CHANNELS=" + support::cpp11::to_string(_output->info()->tensor_shape().z()));
279  build_opts.add_option("-DDEPTH_MULTIPLIER=" + support::cpp11::to_string(depth_multiplier));
280  build_opts.add_option("-DCONV_STRIDE_X=" + support::cpp11::to_string(_conv_stride_x));
281  build_opts.add_option("-DDILATION_X=" + support::cpp11::to_string(dilation.x()));
282  build_opts.add_option("-DDILATION_Y=" + support::cpp11::to_string(dilation.y()));
283  build_opts.add_option_if(_biases != nullptr, "-DHAS_BIAS");
284 
285  if(_is_quantized)
286  {
287  const UniformQuantizationInfo iq_info = _input->info()->quantization_info().uniform();
288  const UniformQuantizationInfo wq_info = _weights->info()->quantization_info().uniform();
289  const UniformQuantizationInfo oq_info = _output->info()->quantization_info().uniform();
290 
291  const bool is_quantized_per_channel = is_data_type_quantized_per_channel(weights->info()->data_type());
292  const bool is_dot8_supported = dot8_supported(CLKernelLibrary::get().get_device()) && !is_quantized_per_channel;
293  build_opts.add_option("-DCONV_STRIDE_Y=" + support::cpp11::to_string(_conv_stride_y));
294  build_opts.add_option("-DINPUT_OFFSET=" + support::cpp11::to_string(-iq_info.offset));
295  build_opts.add_option("-DWEIGHTS_OFFSET=" + support::cpp11::to_string(-wq_info.offset));
296  build_opts.add_option("-DOUTPUT_OFFSET=" + support::cpp11::to_string(oq_info.offset));
297  build_opts.add_option("-DK_OFFSET=" + support::cpp11::to_string(9 * iq_info.offset * wq_info.offset));
298  build_opts.add_option_if(is_quantized_per_channel, "-DPER_CHANNEL_QUANTIZATION");
299  build_opts.add_option_if(is_dot8_supported, "-DIS_DOT8");
300 
301  // Compute non-per-channel multiplier and shift anyway to make OpenCL kernel simpler
302  float multiplier = iq_info.scale * wq_info.scale / oq_info.scale;
303  int output_multiplier = 0;
304  int output_shift = 0;
305  quantization::calculate_quantized_multiplier(multiplier, &output_multiplier, &output_shift);
306  build_opts.add_option("-DOUTPUT_MULTIPLIER=" + support::cpp11::to_string(output_multiplier));
307  build_opts.add_option("-DOUTPUT_SHIFT=" + support::cpp11::to_string(output_shift));
308 
309  if(act_info.enabled())
310  {
311  const int a_val = quantize_qasymm8(act_info.a(), oq_info);
312  const int b_val = quantize_qasymm8(act_info.b(), oq_info);
313  const int o1 = oq_info.offset;
314 
315  build_opts.add_option("-DA_VAL=" + support::cpp11::to_string(a_val));
316  build_opts.add_option("-DB_VAL=" + support::cpp11::to_string(b_val));
317  build_opts.add_option("-DCONST_0=" + support::cpp11::to_string(o1));
318 
319  const float s1 = iq_info.scale;
320  build_opts.add_option("-DS1_VAL=" + float_to_string_with_full_precision(s1));
321  build_opts.add_option("-DO1_VAL=" + support::cpp11::to_string(o1));
322  }
323 
324  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
325  build_opts.add_option("-DWEIGHTS_TYPE=" + get_cl_type_from_data_type(weights->info()->data_type()));
326  build_opts.add_option("-DWEIGHTS_PROMOTED_TYPE=" + get_cl_promoted_type_from_data_type(weights->info()->data_type()));
327  }
328  else
329  {
330  build_opts.add_option_if(act_info.enabled(), "-DA_VAL=" + float_to_string_with_full_precision(act_info.a()));
331  build_opts.add_option_if(act_info.enabled(), "-DB_VAL=" + float_to_string_with_full_precision(act_info.b()));
332  build_opts.add_option_if(act_info.enabled(), "-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
333  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(win_config.second.x().step()));
334  }
335 
336  build_opts.add_option_if(input->info()->data_type() == DataType::F16, "-DIS_F16");
337  build_opts.add_option_if(input->info()->data_type() == DataType::F32, "-DIS_F32");
338 
339  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
340 
341  // Set config_id for enabling LWS tuning
342  _config_id = kernel_name;
343  _config_id += "_";
344  _config_id += lower_string(string_from_data_type(input->info()->data_type()));
345  _config_id += "_";
346  _config_id += support::cpp11::to_string(input->info()->dimension(0));
347  _config_id += "_";
348  _config_id += support::cpp11::to_string(input->info()->dimension(1));
349  _config_id += "_";
350  _config_id += support::cpp11::to_string(input->info()->dimension(2));
351  _config_id += "_";
352  _config_id += support::cpp11::to_string(output->info()->dimension(0));
353  _config_id += "_";
354  _config_id += support::cpp11::to_string(output->info()->dimension(1));
355 }

370 void CLDepthwiseConvolutionLayer3x3NCHWKernel::run(const Window &window, cl::CommandQueue &queue)
371 {
372  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
373  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IKernel::window(), window);
374 
375  Window collapsed = window.collapse_if_possible(ICLKernel::window(), Window::DimZ);
376 
377  // Create input window and adjust
378  Window collapsed_in = collapsed;
379  collapsed_in.adjust(Window::DimX, -_conv_pad_left, true);
380  collapsed_in.adjust(Window::DimY, -_conv_pad_top, true);
381  collapsed_in.set_dimension_step(Window::DimX, collapsed_in.x().step() * _conv_stride_x);
382  collapsed_in.set_dimension_step(Window::DimY, collapsed_in.y().step() * _conv_stride_y);
383 
384  Window slice_in = collapsed_in.first_slice_window_3D();
385  Window slice_out = collapsed.first_slice_window_3D();
386  Window slice_weights = window.first_slice_window_3D();
387  slice_weights.set_dimension_step(Window::DimX, 0);
388  slice_weights.set_dimension_step(Window::DimY, 0);
389 
390  unsigned int idx = 3 * num_arguments_per_3D_tensor();
391 
392  // Set output multipliers in case of quantized data type
393  if(_is_quantized)
394  {
395  Window slice;
396  slice.use_tensor_dimensions(_output_multipliers->info()->tensor_shape());
397  add_1D_tensor_argument(idx, _output_multipliers, slice);
398  add_1D_tensor_argument(idx, _output_shifts, slice);
399  }
400 
401  // Set biases
402  if(_biases != nullptr)
403  {
404  Window slice_biases;
405  slice_biases.use_tensor_dimensions(_biases->info()->tensor_shape());
406  add_1D_tensor_argument(idx, _biases, slice_biases);
407  }
408 
409  do
410  {
411  idx = 0;
412  add_3D_tensor_argument(idx, _input, slice_in);
413  add_3D_tensor_argument(idx, _output, slice_out);
414  add_3D_tensor_argument(idx, _weights, slice_weights);
415 
416  enqueue(queue, *this, slice_out, lws_hint());
417  }
418  while(collapsed.slide_window_slice_3D(slice_out) && collapsed_in.slide_window_slice_3D(slice_in));
419 }
