50 void CLScharr3x3Kernel::configure(const ICLTensor *input, ICLTensor *output_x, ICLTensor *output_y, bool border_undefined)
51 {
52  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::U8);
53  ARM_COMPUTE_ERROR_ON((output_x == nullptr) && (output_y == nullptr));
54 
55  _run_scharr_x = output_x != nullptr;
56  _run_scharr_y = output_y != nullptr;
57 
58  if(_run_scharr_x)
59  {
60  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output_x, 1, DataType::S16);
61  }
62 
63  if(_run_scharr_y)
64  {
65  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output_y, 1, DataType::S16);
66  }
67 
68  _input = input;
69  _output_x = output_x;
70  _output_y = output_y;
71 
72  // Set build options
73  std::set<std::string> build_opts;
74 
75  if(_run_scharr_x)
76  {
77  build_opts.insert("-DGRAD_X");
78  }
79 
80  if(_run_scharr_y)
81  {
82  build_opts.insert("-DGRAD_Y");
83  }
84 
85  // Create kernel
86  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("scharr3x3", build_opts));
87 
88  // Configure kernel window
89  constexpr unsigned int num_elems_processed_per_iteration = 8;
90  constexpr unsigned int num_elems_read_per_iteration = 16;
91  constexpr unsigned int num_elems_written_per_iteration = 8;
92  constexpr unsigned int num_rows_read_per_iteration = 3;
93 
94  Window win = calculate_max_window(*input->info(), Steps(num_elems_processed_per_iteration), border_undefined, border_size());
95 
96  AccessWindowRectangle input_access(input->info(), -border_size().left, -border_size().top, num_elems_read_per_iteration, num_rows_read_per_iteration);
97  AccessWindowHorizontal output_x_access(output_x == nullptr ? nullptr : output_x->info(), 0, num_elems_written_per_iteration);
98  AccessWindowHorizontal output_y_access(output_y == nullptr ? nullptr : output_y->info(), 0, num_elems_written_per_iteration);
99 
100  update_window_and_padding(win, input_access, output_x_access, output_y_access);
101 
102  output_x_access.set_valid_region(win, input->info()->valid_region(), border_undefined, border_size());
103  output_y_access.set_valid_region(win, input->info()->valid_region(), border_undefined, border_size());
104 
105  ICLKernel::configure_internal(win);
106 }

108 void CLScharr3x3Kernel::run(const Window &window, cl::CommandQueue &queue)
109 {
110  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
111  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
112 
113  Window slice = window.first_slice_window_2D();
114  do
115  {
116  unsigned int idx = 0;
117  add_2D_tensor_argument(idx, _input, slice);
118  add_2D_tensor_argument_if((_run_scharr_x), idx, _output_x, slice);
119  add_2D_tensor_argument_if((_run_scharr_y), idx, _output_y, slice);
120 
121  enqueue(queue, *this, slice, lws_hint());
122  }
123  while(window.slide_window_slice_2D(slice));
124 }
