48 void NEDilateKernel::configure(const ITensor *input, ITensor *output, bool border_undefined)
49 {

53  constexpr unsigned int num_elems_processed_per_iteration = 8;
54  constexpr unsigned int num_elems_read_per_iteration = 16;
55  constexpr unsigned int num_elems_written_per_iteration = 8;
56  constexpr unsigned int num_rows_read_per_iteration = 3;
57 
58  // Configure kernel window
59  Window win = calculate_max_window(*input->info(), Steps(num_elems_processed_per_iteration), border_undefined, border_size());

70 void NEDilateKernel::run(const Window &window, const ThreadInfo &info)
71 {
72  ARM_COMPUTE_UNUSED(info);
73  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
74  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(INESimpleKernel::window(), window);
75 
76  Iterator in(_input, window);
77  Iterator out(_output, window);
78 
79  const size_t in_stride = _input->info()->strides_in_bytes()[1];
80 
81  execute_window_loop(window, [&](const Coordinates &)
82  {
83  uint8_t *in_ptr = in.ptr() - 1;
84  const uint8x16_t top_data = vld1q_u8(in_ptr - in_stride);
85  const uint8x16_t mid_data = vld1q_u8(in_ptr);
86  const uint8x16_t bot_data = vld1q_u8(in_ptr + in_stride);
87 
88  uint8x8_t top_high_data = vget_high_u8(top_data);
89  uint8x8_t top_low_data = vget_low_u8(top_data);
90 
91  uint8x8_t mid_high_data = vget_high_u8(mid_data);
92  uint8x8_t mid_low_data = vget_low_u8(mid_data);
93 
94  uint8x8_t bot_high_data = vget_high_u8(bot_data);
95  uint8x8_t bot_low_data = vget_low_u8(bot_data);
96 
97  uint8x8_t p0;
98  uint8x8_t p1;
99 
100  p0 = top_low_data;
101  p1 = vext_u8(top_low_data, top_high_data, 1);
102  p0 = vmax_u8(p0, p1);
103 
104  p1 = vext_u8(top_low_data, top_high_data, 2);
105  p0 = vmax_u8(p0, p1);
106 
107  p1 = mid_low_data;
108  p0 = vmax_u8(p0, p1);
109 
110  p1 = vext_u8(mid_low_data, mid_high_data, 1);
111  p0 = vmax_u8(p0, p1);
112 
113  p1 = vext_u8(mid_low_data, mid_high_data, 2);
114  p0 = vmax_u8(p0, p1);
115 
116  p1 = bot_low_data;
117  p0 = vmax_u8(p0, p1);
118 
119  p1 = vext_u8(bot_low_data, bot_high_data, 1);
120  p0 = vmax_u8(p0, p1);
121 
122  p1 = vext_u8(bot_low_data, bot_high_data, 2);
123  p0 = vmax_u8(p0, p1);
124 
125  vst1_u8(out.ptr(), p0);
126  },
127  in, out);
128 }
