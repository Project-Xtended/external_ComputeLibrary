106 bool Program::build(const cl::Program &program, const std::string &build_options)
107 {
108  try
109  {
110  return program.build(build_options.c_str()) == CL_SUCCESS;
111  }
112  catch(const cl::Error &e)
113  {
114  cl_int err = CL_SUCCESS;
115  const auto build_info = program.getBuildInfo<CL_PROGRAM_BUILD_LOG>(&err);
116 
117  for(auto &pair : build_info)
118  {
119  std::cerr << pair.second << std::endl;
120  }
121 
122  return false;
123  }
124 }

1078 Kernel CLKernelLibrary::create_kernel(const std::string &kernel_name, const StringSet &build_options_set) const
1079 {
1080  // Find which program contains the kernel
1081  auto kernel_program_it = _kernel_program_map.find(kernel_name);
1082 
1083  if(_kernel_program_map.end() == kernel_program_it)
1084  {
1085  ARM_COMPUTE_ERROR_VAR("Kernel %s not found in the CLKernelLibrary", kernel_name.c_str());
1086  }
1087  std::string concat_str;
1088 
1089 #if defined(ARM_COMPUTE_DEBUG_ENABLED)
1090  // Enable debug properties in CL kernels
1091  concat_str += " -DARM_COMPUTE_DEBUG_ENABLED";
1092 #endif // defined(ARM_COMPUTE_DEBUG_ENABLED)
1093 
1094  GPUTarget gpu_arch = get_arch_from_target(get_target_from_device(_device));
1095  concat_str += " -DGPU_ARCH=" + support::cpp11::to_string(
1096  static_cast<std::underlying_type<GPUTarget>::type>(gpu_arch));
1097  if(fp16_supported())
1098  {
1099  concat_str += " -DARM_COMPUTE_OPENCL_FP16_ENABLED=1 ";
1100  }
1101 
1102  if(dot8_supported(_device))
1103  {
1104  concat_str += " -DARM_COMPUTE_OPENCL_DOT8_ENABLED=1 ";
1105  }
1106 
1107  if(dot8_acc_supported(_device))
1108  {
1109  concat_str += " -DARM_COMPUTE_OPENCL_DOT8_ACC_ENABLED=1 ";
1110  }
1111 
1112  if(get_cl_version(_device) == CLVersion::CL20)
1113  {
1114  concat_str += " -cl-std=CL2.0 ";
1115  }
1116  else if(arm_non_uniform_workgroup_supported(_device))
1117  {
1118  concat_str += " -cl-arm-non-uniform-work-group-size ";
1119  }
1120  else
1121  {
1122  ARM_COMPUTE_ERROR("Non uniform workgroup size is not supported!!");
1123  }
1124 
1125  // Check if the program has been built before with same build options.
1126  const std::string program_name = kernel_program_it->second;
1127  const std::string build_options = stringify_set(build_options_set) + concat_str;
1128 
1129  const std::string built_program_name = program_name + "_" + build_options;
1130  auto built_program_it = _built_programs_map.find(built_program_name);
1131 
1132  cl::Program cl_program;
1133 
1134  if(_built_programs_map.end() != built_program_it)
1135  {
1136  // If program has been built, retrieve to create kernel from it
1137  cl_program = built_program_it->second;
1138  }
1139  else
1140  {
1141  // Get program
1142  Program program = load_program(program_name);
1143 
1144  // Build program
1145  cl_program = program.build(build_options);
1146 
1147  // Add built program to internal map
1148  _built_programs_map.emplace(built_program_name, cl_program);
1149  }
1150 
1151  // Create and return kernel
1152  return Kernel(kernel_name, cl_program);
1153 }

1214 {

1216

1218  {
1219  return program_it->second;
1220  }

1223 
1224 #ifdef EMBEDDED_KERNELS
1225  const auto program_source_it = _program_source_map.find(program_name);
1226 
1227  if(_program_source_map.end() == program_source_it)
1228  {
1229  ARM_COMPUTE_ERROR_VAR("Embedded program for %s does not exist.", program_name.c_str());
1230  }
1231 
1232  program = Program(_context, program_name, program_source_it->second);
1233 #else /* EMBEDDED_KERNELS */

1235  std::string source_name = _kernel_path + program_name;
1236  std::string binary_name = source_name + "bin";
1237 
1238  if(std::ifstream(binary_name).is_open())
1239  {
1240  const std::string program_binary = read_file(binary_name, true);
1241  program = Program(_context, _device, program_name, std::vector<unsigned char>(program_binary.begin(), program_binary.end()));
1242  }
1243  else if(std::ifstream(source_name).is_open())
1244  {
1245  program = Program(_context, program_name, read_file(source_name, false));
1246  }
1247  else
1248  {
1249  ARM_COMPUTE_ERROR_VAR("Kernel file %s does not exist.", source_name.c_str());
1250  }
1251 #endif /* EMBEDDED_KERNELS */
1252 
1253  // Insert program to program map
1254  const auto new_program = _programs_map.emplace(program_name, std::move(program));
1255
