70 void NECumulativeDistributionKernel::run(const Window &window, const ThreadInfo &info)
71 {
72  ARM_COMPUTE_UNUSED(info);
73  ARM_COMPUTE_UNUSED(window);
74  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
75  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(INEKernel::window(), window);
76  ARM_COMPUTE_ERROR_ON(_distribution->buffer() == nullptr);
77  ARM_COMPUTE_ERROR_ON(_cumulative_sum->buffer() == nullptr);
78  ARM_COMPUTE_ERROR_ON(_output->buffer() == nullptr);
79  ARM_COMPUTE_ERROR_ON_MSG(_distribution->num_bins() < 256, "Distribution must have 256 bins");
80 
81  // Calculate the cumulative distribution (summed histogram).
82  const uint32_t *hist = _distribution->buffer();
83  uint32_t *cumulative_sum = _cumulative_sum->buffer();
84  uint8_t *output = _output->buffer();
85 
86  // Calculate cumulative distribution
87  std::partial_sum(hist, hist + _histogram_size, cumulative_sum);
88 
89  // Get the number of pixels that have the lowest value in the input image
90  const uint32_t cd_min = *std::find_if(hist, hist + _histogram_size, [](const uint32_t &v)
91  {
92  return v > 0;
93  });
94  const uint32_t image_size = cumulative_sum[_histogram_size - 1];
95 
96  ARM_COMPUTE_ERROR_ON(cd_min > image_size);
97 
98  // Create mapping lookup table
99  if(image_size == cd_min)
100  {
101  std::iota(output, output + _histogram_size, 0);
102  }
103  else
104  {
105  const float diff = image_size - cd_min;
106 
107  for(unsigned int x = 0; x < _histogram_size; ++x)
108  {
109  output[x] = lround((cumulative_sum[x] - cd_min) / diff * 255.0f);
110  }
111  }
112 }
