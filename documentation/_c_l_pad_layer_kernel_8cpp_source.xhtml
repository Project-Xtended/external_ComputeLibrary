33 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const PaddingList &padding, PixelValue constant_value, PaddingMode mode)
34 {
35  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
36  ARM_COMPUTE_UNUSED(constant_value);
37  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
38  ARM_COMPUTE_RETURN_ERROR_ON(padding.size() > input->num_dimensions());
39  if(mode == PaddingMode::REFLECT || mode == PaddingMode::SYMMETRIC)
40  {
41  ARM_COMPUTE_RETURN_ERROR_ON(padding.size() > 3);
42 
43  const auto is_reflect = static_cast<unsigned int>(mode == PaddingMode::REFLECT);
44  for(size_t i = 0; i < padding.size(); ++i)
45  {
46  ARM_COMPUTE_RETURN_ERROR_ON(padding.at(i).first > (input->dimension(i) - is_reflect));
47  ARM_COMPUTE_RETURN_ERROR_ON(padding.at(i).second > (input->dimension(i) - is_reflect));
48  }
49  }
50 
51  if(output->total_size() > 0)
52  {
53  TensorShape padded_shape = misc::shape_calculator::compute_padded_shape(input->tensor_shape(), padding);
54 
55  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(output, input);
56  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), padded_shape);
57  }
58 
59  return Status{};
60 }

62 std::pair<Status, Window> validate_and_configure_window(ITensorInfo *input, ITensorInfo *output, const PaddingList &padding, PixelValue constant_value, PaddingMode mode,
63  unsigned int &num_elems_processed_per_iteration)
64 {
65  ARM_COMPUTE_UNUSED(constant_value, mode);
66 
67  const TensorShape padded_shape = misc::shape_calculator::compute_padded_shape(input->tensor_shape(), padding);
68  auto_init_if_empty(*output, input->clone()->set_tensor_shape(padded_shape));
69 
70  num_elems_processed_per_iteration = std::min(16U, 32U / static_cast<unsigned int>(element_size_from_data_type(input->data_type())));
71  if(input->dimension(0) < num_elems_processed_per_iteration)
72  {
73  num_elems_processed_per_iteration = 1 << static_cast<unsigned int>(std::log2(input->dimension(0)));
74  }
75 
76  // Configure kernel window
77  Window win = calculate_max_window(*output, Steps(num_elems_processed_per_iteration));
78 
79  const int input_start_x = mode == PaddingMode::CONSTANT ? -(padding.at(0).first % num_elems_processed_per_iteration) : 0;
80  const int input_start_y = (mode == PaddingMode::CONSTANT && padding.size() > 1) ? -padding.at(1).first : 0;
81 
82  AccessWindowRectangle input_access(input, input_start_x, input_start_y, num_elems_processed_per_iteration, 1);
83  AccessWindowHorizontal output_access(output, 0, num_elems_processed_per_iteration);
84 
85  const bool window_changed = update_window_and_padding(win, input_access, output_access);
86  output_access.set_valid_region(win, ValidRegion(Coordinates(), output->tensor_shape()));
87 
88  Status err = (window_changed) ? ARM_COMPUTE_CREATE_ERROR(ErrorCode::RUNTIME_ERROR, "Insufficient Padding!") : Status{};
89  return std::make_pair(err, win);
90 }

98 void CLPadLayerKernel::configure(const ICLTensor *input, ICLTensor *output, const PaddingList &padding, PixelValue constant_value, PaddingMode mode)
99 {
100  // Perform validation step
101  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
102  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), padding, constant_value, mode));
103 
104  _input = input;
105  _output = output;
106  _4d_enabled = (mode == PaddingMode::CONSTANT) && (padding.size() > 3);
107 
108  // Configure window
109  unsigned int vec_size;
110  auto win_config = validate_and_configure_window(input->info(), output->info(), padding, constant_value, mode, vec_size);
111  ARM_COMPUTE_ERROR_THROW_ON(win_config.first);
112  ICLKernel::configure_internal(win_config.second);
113 
114  // Set build options
115  std::string kernel_name = "pad_layer_";
116 
117  const DataType &data_type = input->info()->data_type();
118  const unsigned int input_width = input->info()->dimension(0);
119  const unsigned int input_height = input->info()->dimension(1);
120  const unsigned int input_depth = input->info()->dimension(2);
121  const unsigned int pad_x_before = padding.at(0).first;
122  const unsigned int pad_y_before = padding.size() > 1 ? padding.at(1).first : 0;
123  const unsigned int pad_z_before = padding.size() > 2 ? padding.at(2).first : 0;
124  const unsigned int pad_right_start = input_width + pad_x_before;
125 
126  _input_start_x = mode == PaddingMode::CONSTANT ? -(pad_x_before % vec_size) : 0;
127  _input_start_y = (mode == PaddingMode::CONSTANT && padding.size() > 1) ? -padding.at(1).first : 0;
128 
129  CLBuildOptions build_opts;
130  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
131  build_opts.add_option("-DSELECT_DT=" + get_cl_select_type_from_data_type(data_type));
132  build_opts.add_option("-DVEC_SIZE=" + support::cpp11::to_string(vec_size));
133  build_opts.add_option("-DPAD_X_BEFORE=" + support::cpp11::to_string(pad_x_before));
134  build_opts.add_option("-DSRC_WIDTH=" + support::cpp11::to_string(input_width));
135  if(padding.size() > 1)
136  {
137  build_opts.add_option("-DPAD_Y_BEFORE=" + support::cpp11::to_string(pad_y_before));
138  build_opts.add_option("-DSRC_HEIGHT=" + support::cpp11::to_string(input_height));
139 
140  if(padding.size() > 2)
141  {
142  build_opts.add_option("-DPAD_Z_BEFORE=" + support::cpp11::to_string(pad_z_before));
143  build_opts.add_option("-DSRC_DEPTH=" + support::cpp11::to_string(input_depth));
144  }
145  }
146 
147  switch(mode)
148  {
149  case PaddingMode::CONSTANT:
150  {
151  kernel_name += "constant";
152 
153  build_opts.add_option("-DCONST_VAL=" + string_from_pixel_value(constant_value, data_type));
154  build_opts.add_option_if(pad_x_before >= vec_size, "-DNUM_THREADS_TO_SKIP_X=" + support::cpp11::to_string(pad_x_before / vec_size));
155 
156  if(_4d_enabled)
157  {
158  build_opts.add_option("-DPAD_W_BEFORE=" + support::cpp11::to_string(padding.at(3).first));
159  build_opts.add_option("-DSRC_BATCH=" + support::cpp11::to_string(input->info()->dimension(3)));
160  }
161 
162  break;
163  }
164  case PaddingMode::SYMMETRIC:
165  case PaddingMode::REFLECT:
166  {
167  kernel_name += "symmetric_reflect";
168 
169  const auto is_reflect = static_cast<unsigned int>(mode == PaddingMode::REFLECT);
170 
171  const unsigned int pad_x_before_remainder = pad_x_before % vec_size;
172  const unsigned int pad_x_after_remainder = pad_right_start % vec_size;
173  const unsigned int after_pad_fact_x = (2 * input_width + pad_x_before) - is_reflect;
174  const unsigned int output_last_x = ceil_to_multiple(pad_right_start + padding.at(0).second, vec_size);
175 
176  build_opts.add_option("-DIS_REFLECT=" + support::cpp11::to_string(is_reflect));
177  build_opts.add_option("-DPAD_X_BEFORE_REMAINDER=" + support::cpp11::to_string(pad_x_before_remainder));
178  build_opts.add_option("-DPAD_X_AFTER_REMAINDER=" + support::cpp11::to_string(pad_x_after_remainder));
179  build_opts.add_option("-DPAD_X_BEFORE_REMAINDER_REFL=" + support::cpp11::to_string((pad_x_before_remainder + is_reflect) % vec_size));
180  build_opts.add_option("-DPAD_X_AFTER_REMAINDER_REFL=" + support::cpp11::to_string((pad_x_after_remainder - is_reflect) % vec_size));
181  build_opts.add_option("-DAFTER_PAD_FACT_X=" + support::cpp11::to_string(after_pad_fact_x));
182  build_opts.add_option_if(after_pad_fact_x < output_last_x, "-DAFTER_PAD_REM=" + support::cpp11::to_string(after_pad_fact_x % vec_size));
183 
184  break;
185  }
186  default:
187  ARM_COMPUTE_ERROR("Padding mode not supported.");
188  }
189 
190  // Create kernel
191  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel(kernel_name, build_opts.options()));
192 }

203 void CLPadLayerKernel::run(const Window &window, cl::CommandQueue &queue)
204 {
205  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
206  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(ICLKernel::window(), window);
207 
208  Window win_in = window;
209  win_in.adjust(Window::DimX, _input_start_x, true);
210  win_in.adjust(Window::DimY, _input_start_y, true);
211 
212  Window slice_out = window.first_slice_window_3D();
213  Window slice_in = win_in.first_slice_window_3D();
214  unsigned int batch = 0;
215  do
216  {
217  unsigned int idx = 0;
218  add_3D_tensor_argument(idx, _input, slice_in);
219  add_3D_tensor_argument(idx, _output, slice_out);
220  if(_4d_enabled)
221  {
222  add_argument<unsigned int>(idx, batch++);
223  }
224 
225  enqueue(queue, *this, slice_out, lws_hint());
226  }
227  while(window.slide_window_slice_3D(slice_out) && win_in.slide_window_slice_3D(slice_in));
228 }
