47 void GCAbsoluteDifferenceKernel::configure(const IGCTensor *input1, const IGCTensor *input2, IGCTensor *output)
48 {
49  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input1, 1, DataType::U8);
50  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input2, 1, DataType::U8);
51  ARM_COMPUTE_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(output, 1, DataType::U8);
52  ARM_COMPUTE_ERROR_ON_MISMATCHING_DATA_TYPES(input1, input2, output);
53 
54  _input1 = input1;
55  _input2 = input2;
56  _output = output;
57 
58  constexpr unsigned int num_elems_processed_per_iteration = 4;
59 
60  // Set kernel build options
61  std::set<std::string> build_opts;
62  build_opts.emplace("#define LOCAL_SIZE_X " + support::cpp11::to_string(1));
63  build_opts.emplace("#define LOCAL_SIZE_Y " + support::cpp11::to_string(1));
64  build_opts.emplace("#define LOCAL_SIZE_Z " + support::cpp11::to_string(1));
65 
66  // Create kernel
67  _kernel = static_cast<GCKernel>(GCKernelLibrary::get().create_kernel("absdiff", build_opts));
68 
69  // Configure kernel window
70  Window win = calculate_max_window(*input1->info(), Steps(num_elems_processed_per_iteration));
71 
72  AccessWindowRectangle input1_access(input1->info(), 0, 0, 4, 1);
73  AccessWindowRectangle input2_access(input2->info(), 0, 0, 4, 1);
74  AccessWindowRectangle output_access(output->info(), 0, 0, 4, 1);
75 
76  update_window_and_padding(win, input1_access, input2_access, output_access);
77 
78  ValidRegion valid_region = intersect_valid_regions(input1->info()->valid_region(),
79  input2->info()->valid_region());
80 
81  output_access.set_valid_region(win, valid_region);
82 
83  IGCKernel::configure(win);
84 }

86 void GCAbsoluteDifferenceKernel::run(const Window &window)
87 {
88  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
89  ARM_COMPUTE_ERROR_ON_INVALID_SUBWINDOW(IGCKernel::window(), window);
90 
91  _kernel.use();
92 
93  Window slice = window.first_slice_window_2D();
94  do
95  {
96  unsigned int idx = 0;
97  unsigned int binding = 1; // SSBO binding starts from 1.
98  add_2D_tensor_argument(idx, _input1, binding++, slice);
99  add_2D_tensor_argument(idx, _input2, binding++, slice);
100  add_2D_tensor_argument(idx, _output, binding++, slice);
101 
102  _kernel.update_shader_params();
103 
104  enqueue(*this, slice);
105  }
106  while(window.slide_window_slice_2D(slice));
107 }
