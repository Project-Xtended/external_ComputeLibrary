52 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *output, const PermutationVector &perm)
53 {
54  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
55  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
56  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
57  ARM_COMPUTE_RETURN_ERROR_ON_MSG(input->num_dimensions() < 1 || input->num_dimensions() > 4,
58  "Permutation upto 4-D input tensor is supported");
59  ARM_COMPUTE_RETURN_ERROR_ON_MSG(perm.num_dimensions() < 1 || perm.num_dimensions() > 4,
60  "Permutation vector size should be less than or equal to 4");
61  for(const auto &p : perm)
62  {
63  ARM_COMPUTE_RETURN_ERROR_ON_MSG(p >= perm.num_dimensions(), "Permutation vector has invalid values");
64  }
65 
66  // Validate configured output
67  if(output->total_size() != 0)
68  {
69  const TensorShape output_shape = misc::shape_calculator::compute_permutation_output_shape(*input, perm);
70  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), output_shape);
71  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
72  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
73  }
74  return Status{};
75 }

78 void CLPermuteKernel::configure(const ICLTensor *input, ICLTensor *output, const PermutationVector &perm)
79 {
80  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
81  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(), output->info(), perm));
82 
83  _input = input;
84  _output = output;
85  _perm = perm;
86 
87  const TensorShape output_shape = get_output_shape(input->info(), perm);
88  // Output auto inizialitation if not yet initialized
89  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(output_shape));
90 
91  // Create kernel
92  CLBuildOptions build_opts;
93  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(input->info()->data_type()));
94  build_opts.add_option("-DDEPTH_IN=" + support::cpp11::to_string(input->info()->dimension(2)));
95  // New positions of width(W), height(H), channel(C) and batch(D) based on permutation vector
96  build_opts.add_option("-DP1=" + support::cpp11::to_string((_perm.num_dimensions() >= 1) ? perm[0] : 0));
97  build_opts.add_option("-DP2=" + support::cpp11::to_string((_perm.num_dimensions() >= 2) ? perm[1] : 1));
98  build_opts.add_option("-DP3=" + support::cpp11::to_string((_perm.num_dimensions() >= 3) ? perm[2] : 2));
99  build_opts.add_option("-DP4=" + support::cpp11::to_string((_perm.num_dimensions() >= 4) ? perm[3] : 3));
100 
101  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("permute", build_opts.options()));
102 
103  // Configure kernel window
104  Window win = calculate_max_window(*input->info(), Steps());
105 
106  // The CLPermute doesn't need padding so update_window_and_padding() can be skipped
107  Coordinates coord;
108  coord.set_num_dimensions(output->info()->num_dimensions());
109  output->info()->set_valid_region(ValidRegion(coord, output->info()->tensor_shape()));
110 
111  ICLKernel::configure_internal(win);
112 }

122 void CLPermuteKernel::run(const Window &window, cl::CommandQueue &queue)
123 {
124  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
125  ARM_COMPUTE_ERROR_ON_MISMATCHING_WINDOWS(ICLKernel::window(), window);
126 
127  Window slice_in = window.first_slice_window_4D().collapse(ICLKernel::window(), 2, 4);
128 
129  // Setup output slice
130  Window slice_out(slice_in);
131  slice_out.set(Window::DimX, Window::Dimension(0, 0, 0));
132  slice_out.set(Window::DimY, Window::Dimension(0, 0, 0));
133  slice_out.set(Window::DimZ, Window::Dimension(0, 0, 0));
134  slice_out.set(3, Window::Dimension(0, 0, 0));
135 
136  do
137  {
138  unsigned int idx = 0;
139  add_4D_tensor_argument(idx, _input, slice_in);
140  add_4D_tensor_argument(idx, _output, slice_out);
141  enqueue(queue, *this, slice_in, lws_hint());
142  }
143  while(window.slide_window_slice_4D(slice_in) && window.slide_window_slice_4D(slice_out));
144 }
