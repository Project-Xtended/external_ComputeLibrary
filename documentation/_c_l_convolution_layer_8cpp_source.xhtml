46 void CLConvolutionLayer::configure(ICLTensor *input, const ICLTensor *weights, const ICLTensor *biases, ICLTensor *output, const PadStrideInfo &conv_info, const WeightsInfo &weights_info,
47  const Size2D &dilation, const ActivationLayerInfo &act_info, bool enable_fast_math, unsigned int num_groups)
48 {
49  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
50  ARM_COMPUTE_ERROR_THROW_ON(CLConvolutionLayer::validate(input->info(), weights->info(), ((biases != nullptr) ? biases->info() : nullptr), output->info(), conv_info, weights_info, dilation, act_info,
51  enable_fast_math, num_groups));
52 
53  switch(CLConvolutionLayer::get_convolution_method(input->info(), weights->info(), output->info(), conv_info,
54  weights_info, act_info, CLScheduler::get().target(), dilation, enable_fast_math))
55  {
56  case ConvolutionMethod::WINOGRAD:
57  {
58  ARM_COMPUTE_ERROR_ON(num_groups != 1);
59  auto f = arm_compute::support::cpp14::make_unique<CLWinogradConvolutionLayer>(_memory_manager);
60  f->configure(input, weights, biases, output, conv_info, act_info, enable_fast_math);
61  _function = std::move(f);
62  break;
63  }
64  case ConvolutionMethod::DIRECT:
65  {
66  ARM_COMPUTE_ERROR_ON(num_groups != 1);
67  auto f = arm_compute::support::cpp14::make_unique<CLDirectConvolutionLayer>();
68  f->configure(input, weights, biases, output, conv_info, act_info);
69  _function = std::move(f);
70  break;
71  }
72  case ConvolutionMethod::GEMM:
73  {
74  auto f = arm_compute::support::cpp14::make_unique<CLGEMMConvolutionLayer>(_memory_manager);
75  f->configure(input, weights, biases, output, conv_info, weights_info, dilation, act_info, num_groups);
76  _function = std::move(f);
77  break;
78  }
79  case ConvolutionMethod::FFT:
80  {
81  auto f = arm_compute::support::cpp14::make_unique<CLFFTConvolutionLayer>(_memory_manager);
82  f->configure(input, weights, biases, output, conv_info, act_info);
83  _function = std::move(f);
84  break;
85  }
86  default:
87  ARM_COMPUTE_ERROR("Not supported.");
88  break;
89  }
90 }

92 Status CLConvolutionLayer::validate(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *biases, const ITensorInfo *output, const PadStrideInfo &conv_info,
93  const WeightsInfo &weights_info, const Size2D &dilation, const ActivationLayerInfo &act_info, bool enable_fast_math, unsigned int num_groups)
94 {
95  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, weights, output);
96  ARM_COMPUTE_RETURN_ERROR_ON_MSG((num_groups != 1) && (input->data_layout() != DataLayout::NCHW), "Grouping (num_groups != 1) with NHWC data layout is not supported");
97 
98  const GPUTarget gpu_target = CLScheduler::get().target();
99 
100  switch(CLConvolutionLayer::get_convolution_method(input, weights, output, conv_info, weights_info, act_info, gpu_target, dilation, enable_fast_math))
101  {
102  case ConvolutionMethod::WINOGRAD:
103  {
104  //Validate Winograd
105  ARM_COMPUTE_RETURN_ERROR_ON_MSG(num_groups != 1, "Grouping (num_groups != 1) with CLWinogradConvolutionLayer is not supported");
106  ARM_COMPUTE_RETURN_ON_ERROR(CLWinogradConvolutionLayer::validate(input, weights, biases, output, conv_info, act_info, enable_fast_math));
107  break;
108  }
109  case ConvolutionMethod::DIRECT:
110  {
111  // Validate direct convolution layer
112  ARM_COMPUTE_RETURN_ERROR_ON_MSG(num_groups != 1, "Grouping (num_groups != 1) with CLDirectConvolutionLayer is not supported");
113  ARM_COMPUTE_RETURN_ON_ERROR(CLDirectConvolutionLayer::validate(input, weights, biases, output, conv_info, act_info));
114  break;
115  }
116  case ConvolutionMethod::GEMM:
117  {
118  // Validate gemm-based convolution layer
119  ARM_COMPUTE_RETURN_ON_ERROR(CLGEMMConvolutionLayer::validate(input, weights, biases, output, conv_info, weights_info, dilation, act_info, num_groups));
120  break;
121  }
122  case ConvolutionMethod::FFT:
123  {
124  // Validate FFT-based convolution layer
125  ARM_COMPUTE_RETURN_ON_ERROR(CLFFTConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info));
126  break;
127  }
128  default:
129  ARM_COMPUTE_ERROR("Not supported.");
130  break;
131  }
132 
133  return Status{};
134 }

136 ConvolutionMethod CLConvolutionLayer::get_convolution_method(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *output, const PadStrideInfo &conv_info,
137  const WeightsInfo &weights_info, const ActivationLayerInfo &act_info, const GPUTarget gpu_target, const Size2D &dilation, bool enable_fast_math)
138 {
139  ARM_COMPUTE_ERROR_ON_NULLPTR(input);
140  ARM_COMPUTE_ERROR_ON_NULLPTR(output);
141  ARM_COMPUTE_ERROR_ON_NULLPTR(weights);
142  ARM_COMPUTE_UNUSED(weights_info);
143  ARM_COMPUTE_UNUSED(gpu_target);
144 
145  const size_t idx_w = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::WIDTH);
146  const size_t idx_h = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::HEIGHT);
147  const size_t idx_c = get_data_layout_dimension_index(input->data_layout(), DataLayoutDimension::CHANNEL);
148 
149  /* Input spatial dims, kernel size, IFM/OFM, conv info*/
150  using ConvolutionConfiguration = std::tuple<Size2D, Size2D, Size2D, PadStrideInfo, DataLayout>;
151  using ConfigurationMethod = std::pair<ConvolutionConfiguration, ConvolutionMethod>;
152 
153  const std::vector<ConfigurationMethod> known_configs =
154  {
155  // Alexnet
156  ConfigurationMethod(ConvolutionConfiguration(Size2D(27U, 27U), Size2D(5U, 5U), Size2D(48U, 128U), PadStrideInfo(1U, 1U, 2U, 2U), DataLayout::NCHW), ConvolutionMethod::DIRECT),
157  // VGG16 / VGG19
158  ConfigurationMethod(ConvolutionConfiguration(Size2D(224U, 224U), Size2D(3U, 3U), Size2D(3U, 64U), PadStrideInfo(1U, 1U, 1U, 1U), DataLayout::NCHW), ConvolutionMethod::DIRECT),
159  // Mobilenet 224
160  ConfigurationMethod(ConvolutionConfiguration(Size2D(224U, 224U), Size2D(3U, 3U), Size2D(3U, 32U), PadStrideInfo(2U, 2U, 0U, 1U, 0U, 1U, DimensionRoundingType::FLOOR), DataLayout::NCHW), ConvolutionMethod::GEMM),
161  // Mobilenet 160
162  ConfigurationMethod(ConvolutionConfiguration(Size2D(160U, 160U), Size2D(3U, 3U), Size2D(3U, 24U), PadStrideInfo(2U, 2U, 0U, 1U, 0U, 1U, DimensionRoundingType::FLOOR), DataLayout::NCHW), ConvolutionMethod::GEMM),
163  // Mobilenet 224
164  ConfigurationMethod(ConvolutionConfiguration(Size2D(224U, 224U), Size2D(3U, 3U), Size2D(3U, 32U), PadStrideInfo(2U, 2U, 0U, 1U, 0U, 1U, DimensionRoundingType::FLOOR), DataLayout::NHWC), ConvolutionMethod::GEMM),
165  // Mobilenet 160
166  ConfigurationMethod(ConvolutionConfiguration(Size2D(160U, 160U), Size2D(3U, 3U), Size2D(3U, 24U), PadStrideInfo(2U, 2U, 0U, 1U, 0U, 1U, DimensionRoundingType::FLOOR), DataLayout::NHWC), ConvolutionMethod::GEMM),
167  };
168 
169  const auto find_config = [&](ConfigurationMethod c)
170  {
171  const ConvolutionConfiguration config = c.first;
172  const PadStrideInfo info = std::get<3>(config);
173  const DataLayout data_layout = std::get<4>(config);
174 
175  return std::get<0>(config) == Size2D(input->dimension(idx_w), input->dimension(idx_h)) && std::get<1>(config) == Size2D(weights->dimension(idx_w), weights->dimension(idx_h))
176  && std::get<2>(config) == Size2D(weights->dimension(idx_c), weights->dimension(3)) && info.pad_top() == conv_info.pad_top() && info.pad_right() == conv_info.pad_right()
177  && info.pad_bottom() == conv_info.pad_bottom() && info.pad_left() == conv_info.pad_left() && info.stride() == conv_info.stride() && (data_layout == input->data_layout());
178  };
179 
180  std::vector<ConfigurationMethod>::const_iterator found;
181  if((found = std::find_if(known_configs.begin(), known_configs.end(), find_config)) != known_configs.end())
182  {
183  return (*found).second;
184  }
185 
186  if(dilation != Size2D(1U, 1U))
187  {
188  return ConvolutionMethod::GEMM;
189  }
190  else
191  {
192  // SRGAN
193  if((input->dimension(idx_h) > 720U) && (output->dimension(idx_h) > 720U) && (weights->dimension(idx_h) == 9) && (conv_info.pad_top() < 3)
194  && (CLDirectConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info)))
195  {
196  return ConvolutionMethod::DIRECT;
197  }
198  if((weights->dimension(idx_h) > 7) && (input->dimension(idx_c) > output->dimension(idx_c)) && (CLFFTConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info)))
199  {
200  return ConvolutionMethod::FFT;
201  }
202  if(input->dimension(idx_c) < 16)
203  {
204  return ConvolutionMethod::GEMM;
205  }
206  return bool(CLWinogradConvolutionLayer::validate(input, weights, nullptr, output, conv_info, act_info, enable_fast_math)) ? ConvolutionMethod::WINOGRAD : ConvolutionMethod::GEMM;
207  }
208 }
