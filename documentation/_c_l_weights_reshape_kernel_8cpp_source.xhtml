42 Status validate_arguments(const ITensorInfo *input, const ITensorInfo *biases, const ITensorInfo *output, unsigned int num_groups)
43 {
44  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, output);
45  ARM_COMPUTE_RETURN_ERROR_ON_F16_UNSUPPORTED(input);
46  ARM_COMPUTE_RETURN_ERROR_ON(input->data_type() == DataType::UNKNOWN);
47  ARM_COMPUTE_RETURN_ERROR_ON(num_groups == 0);
48  ARM_COMPUTE_RETURN_ERROR_ON(input->data_layout() == DataLayout::NHWC && num_groups > 1);
49  ARM_COMPUTE_RETURN_ERROR_ON(input->num_dimensions() > 4 && num_groups > 1);
50  ARM_COMPUTE_RETURN_ERROR_ON((input->dimension(3) % num_groups) != 0);
51 
52  if(biases != nullptr)
53  {
54  ARM_COMPUTE_RETURN_ERROR_ON(!is_data_type_float(input->data_type()));
55  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, biases);
56  ARM_COMPUTE_RETURN_ERROR_ON((input->num_dimensions() == 4) && (biases->num_dimensions() != 1));
57  ARM_COMPUTE_RETURN_ERROR_ON((input->num_dimensions() == 5) && (biases->num_dimensions() != 2));
58  ARM_COMPUTE_RETURN_ERROR_ON((input->num_dimensions() == 4) && (biases->dimension(0) != input->tensor_shape()[3]));
59  ARM_COMPUTE_RETURN_ERROR_ON((input->num_dimensions() == 5) && (biases->dimension(0) != input->tensor_shape()[3] || biases->dimension(1) != input->tensor_shape()[4]));
60  }
61 
62  // Checks performed when output is configured
63  if(output->total_size() != 0)
64  {
65  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DIMENSIONS(output->tensor_shape(), compute_weights_reshaped_shape(*input, biases != nullptr, num_groups));
66  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
67  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_QUANTIZATION_INFO(input, output);
68  }
69 
70  return Status{};
71 }

79 void CLWeightsReshapeKernel::configure(const ICLTensor *input, const ICLTensor *biases, ICLTensor *output, unsigned int num_groups)
80 {
81  ARM_COMPUTE_ERROR_ON_NULLPTR(input, output);
82 
83  // Output tensor auto inizialitation if not yet initialized
84  auto_init_if_empty(*output->info(), input->info()->clone()->set_tensor_shape(compute_weights_reshaped_shape(*input->info(), (biases != nullptr), num_groups)));
85 
86  // Perform validation step
87  ARM_COMPUTE_ERROR_THROW_ON(validate_arguments(input->info(),
88  (biases != nullptr) ? biases->info() : nullptr,
89  output->info(), num_groups));
90 
91  const DataType data_type = input->info()->data_type();
92 
93  _biases = biases;
94  _output = output;
95  _input = input;
96 
97  // Create build options
98  CLBuildOptions build_opts;
99  build_opts.add_option("-DDATA_TYPE=" + get_cl_type_from_data_type(data_type));
100  build_opts.add_option("-DNUM_GROUPS=" + support::cpp11::to_string(num_groups));
101  build_opts.add_option_if(biases != nullptr, "-DHAS_BIAS");
102 
103  // Create kernel
104  _kernel = static_cast<cl::Kernel>(CLKernelLibrary::get().create_kernel("reshape_to_columns", build_opts.options()));
105 
106  // Configure window
107  Window win = calculate_max_window(*input->info(), Steps());
108  // The CLWeightsReshapeKernel doesn't need padding so update_window_and_padding() can be skipped
109  output->info()->set_valid_region(ValidRegion(Coordinates(), output->info()->tensor_shape()));
110  ICLKernel::configure_internal(win);
111 }

119 void CLWeightsReshapeKernel::run(const Window &window, cl::CommandQueue &queue)
120 {
121  ARM_COMPUTE_ERROR_ON_UNCONFIGURED_KERNEL(this);
122  ARM_COMPUTE_ERROR_ON_MISMATCHING_WINDOWS(ICLKernel::window(), window);
123 
124  Window out_window;
125  out_window.use_tensor_dimensions(_output->info()->tensor_shape());
126 
127  Window in_slice = window.first_slice_window_3D();
128  Window out_slice = out_window.first_slice_window_2D();
129 
130  Window biases_window;
131  Window biases_slice;
132 
133  unsigned int idx = num_arguments_per_3D_tensor() + num_arguments_per_2D_tensor();
134  idx += (_biases != nullptr) ? num_arguments_per_1D_tensor() : 0;
135  _kernel.setArg<cl_uint>(idx++, _input->info()->dimension(0));
136  _kernel.setArg<cl_uint>(idx++, _input->info()->dimension(1));
137  _kernel.setArg<cl_uint>(idx++, _input->info()->dimension(2));
138  _kernel.setArg<cl_uint>(idx++, _input->info()->dimension(3));
139  _kernel.setArg<cl_uint>(idx++, _output->info()->strides_in_bytes().z());
140 
141  if(_biases != nullptr)
142  {
143  biases_window.use_tensor_dimensions(_biases->info()->tensor_shape());
144  biases_slice = biases_window.first_slice_window_1D();
145  }
146 
147  do
148  {
149  // Set arguments
150  unsigned idx = 0;
151  add_3D_tensor_argument(idx, _input, in_slice);
152  add_2D_tensor_argument(idx, _output, out_slice);
153  if(_biases != nullptr)
154  {
155  add_1D_tensor_argument(idx, _biases, biases_slice);
156  ARM_COMPUTE_UNUSED(biases_window.slide_window_slice_1D(biases_slice));
157  }
158 
159  // Run kernel
160  enqueue(queue, *this, in_slice, lws_hint());
161  }
162  while(window.slide_window_slice_4D(in_slice) && out_window.slide_window_slice_2D(out_slice));
163 }
